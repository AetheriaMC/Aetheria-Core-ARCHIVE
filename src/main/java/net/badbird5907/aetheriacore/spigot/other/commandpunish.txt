package net.badbird5907.aetheriacore.spigot.commands;

import net.badbird5907.aetheriacore.spigot.AetheriaCore;
import org.bukkit.Bukkit;
import org.bukkit.ChatColor;
import org.bukkit.Material;
import org.bukkit.command.Command;
import org.bukkit.command.CommandExecutor;
import org.bukkit.command.CommandSender;
import org.bukkit.entity.Player;
import org.bukkit.inventory.Inventory;
import org.bukkit.inventory.ItemStack;
import org.bukkit.inventory.meta.ItemMeta;

import java.util.ArrayList;

public class commandpunish implements CommandExecutor {
    AetheriaCore plugin;

    public commandpunish(AetheriaCore plugin) {
        this.plugin = plugin;
    }

    @Override
    public boolean onCommand(CommandSender sender, Command command,String label, String[] args) {
        if (sender instanceof Player){
            Player player = (Player) sender;

            plugin.openPunishGUI(player);

            ArrayList<Player> player_list = new ArrayList<>(player.getServer().getOnlinePlayers());

//            for (int i = 0; i < player_list.size(); i++){
//                player.sendMessage(player_list.get(i).getDisplayName());  THIS IS TO LOST ALL PLAYERS
            Inventory bangui = Bukkit.createInventory(player, 45, ChatColor.BLUE +"Player List");
            // for every player add name&head to gui
            for (int i = 0;i < player_list.size();i++){
                ItemStack playerHead = new ItemStack(Material.PLAYER_HEAD, 1);
                ItemMeta meta = playerHead.getItemMeta();
                meta.setDisplayName(player_list.get(i).getDisplayName());
                ArrayList<String> lore = new ArrayList<>();
                lore.add(ChatColor.GOLD + "Player Health:" + player_list.get(i).getHealth());
                lore.add(ChatColor.GOLD + "Flight Allowed:" + player_list.get(i).getAllowFlight());
                lore.add(ChatColor.GOLD + "Player Walk Speed:" + player_list.get(i).getWalkSpeed());


                meta.setLore(lore);
                playerHead.setItemMeta(meta);

                bangui.addItem(playerHead);

            }
            player.openInventory(bangui);
        }

        return true;
    }
}
